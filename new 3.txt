function getClients($conn, $search, $limit = 9, $offset = 0) {
    if ($search !== "") {
        $stmt = $conn->prepare("
            SELECT s.id, s.client_name, s.client_code, st.staff_name, s.view_count
            FROM surveys s
            LEFT JOIN staff_list st ON s.user_id = st.id
            WHERE s.client_name LIKE ? OR s.client_code LIKE ?
            ORDER BY s.id DESC
            LIMIT ? OFFSET ?
        ");
        $like = "%$search%";
        $stmt->bind_param("ssii", $like, $like, $limit, $offset);
    } else {
        $stmt = $conn->prepare("
            SELECT s.id, s.client_name, s.client_code, st.staff_name, s.view_count
            FROM surveys s
            LEFT JOIN staff_list st ON s.user_id = st.id
            ORDER BY s.id DESC
            LIMIT ? OFFSET ?
        ");
        $stmt->bind_param("ii", $limit, $offset);
    }

    $stmt->execute();
    $result = $stmt->get_result();
    $clients = [];
    while ($row = $result->fetch_assoc()) {
        $staff = $row['staff_name'] ?? 'غير محدد';
        $clients[] = [
            'id' => $row['id'],
            'client_name' => $row['client_name'] ?: '-',
            'client_code' => $row['client_code'] ?: '-',
            'staff' => $staff,
            'view_count' => $row['view_count'] ?? 0
        ];
    }
    return $clients;
}

<script>
const input = document.getElementById('searchInput');
const tableBody = document.getElementById('clientTable');

function fetchClients(query='', page=1){
    const xhr = new XMLHttpRequest();
    xhr.open('GET', 'pepol.php?ajax=1&search=' + encodeURIComponent(query) + '&page=' + page, true);
    xhr.onload = function(){
        if(this.status === 200){
            tableBody.innerHTML = this.responseText;
        }
    };
    xhr.send();
}

// البحث أثناء الكتابة مباشرة
input.addEventListener('input', () => {
    fetchClients(input.value, 1);
});

// زيادة عدد الزيارات قبل فتح صفحة العميل
function increaseView(clientId){
    fetch('pepol.php?increase_view_id=' + clientId)
    .then(res => res.json())
    .then(data => {
        window.location = 'client.php?id=' + clientId;
    });
}
